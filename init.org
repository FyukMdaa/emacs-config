* setup.el のブートストラップ

twist.nix がすでにすべてのパッケージを Nix store 上にビルド・配置して
いるので、setup.el 標準の =:package= は何もしない =no-op= に置き換える。

#+begin_src emacs-lisp
;; -*- lexical-binding: t; -*-

(eval-and-compile
  (require 'setup))

(setup-define :package
  (lambda (_order) nil)
  :documentation "no-op: パッケージのインストールは twist.nix が担当する。"
  :repeatable t
  :shorthand #'cadr)
#+end_src

setup.elで =setopt= を =:opt= で使えるように
#+begin_src elisp
(setup-define :opt
  (lambda (&rest pairs)
    `(setopt ,@pairs))
  :after-loaded t)
#+end_src
=:load-after= を使えるようにする
#+begin_src elisp
  (setup-define :load-after
  (lambda (&rest features)
    (let ((body `(require ',(setup-get 'feature))))
      (dolist (feature (nreverse features))
        (setopt body `(with-eval-after-load ',feature ,body)))
      body))
  :documentation "Load the current feature after FEATURES.")
#+end_src
=:needs= を使えるようにする
#+begin_src elisp
  (setup-define :needs
  (lambda (executable)
    `(unless (executable-find ,executable)
       ,(setup-quit)))
  :documentation "If EXECUTABLE is not in the path, stop here."
  :repeatable 1)
#+end_src

** vertico
#+begin_src elisp
  (setup (:package vertico)
    (:opt vertico-resize nil
          vertico-count 20
          vertico-cycle t)
    (:with-map vertico-map
      ;; ディレクトリ移動時にディレクトリ単位で移動できる。
      (:bind "DEL" vertico-directory-delete-char))

    (vertico-mode)

    ;; verticoの選択行を指す指アイコンを追加する設定
    (:require nerd-icons)
    (defvar +vertico-current-arrow t)

    (defun +vertico-format-candidate-arrow-advice (orig-fun cand prefix suffix index start)
      "Vertico の選択行に指アイコンを表示するアドバイス"
      (let ((res (funcall orig-fun cand prefix suffix index start)))
        (if (and +vertico-current-arrow 
                 (not (bound-and-true-p vertico-flat-mode)))
            (let ((arrow (nerd-icons-faicon "nf-fa-hand_o_right")))
              (if (bound-and-true-p vertico-grid-mode)
                  (if (= vertico--index index)
                      (concat arrow " " res)
                    (concat "  " res))
                (if (= vertico--index index)
                    (concat " " arrow " " res)
                  (concat "    " res))))
          res)))

    (advice-add 'vertico-format-candidate :around #'+vertico-format-candidate-arrow-advice))
#+end_src

* 補完
** corfu
#+begin_src elisp
  (setup (:package corfu)

  (:option
   corfu-auto t ;; 補完候補を表示する
   corfu-auto-delay 0 ;; 文字列入力してから補完候補が表示されるまでのディレイ
   corfu-popupinfo-delay 0 ;; 補完候補の関数名の横に、さらに説明文がポップアップされるまでのディレイ
   corfu-auto-prefix 1 ;; 文字列入力の何文字目から補完候補を表示するか
   corfu-cycle t ;; 補完候補の一番下までカーソルが行ったら上に戻る
   corfu-preselect 'prompt
   corfu-quit-no-match 'separator)

  (:with-map corfu-map
    (:bind
     ;; 補完時に SPC をセパレータとして orderless 補完できる。
     "SPC" corfu-insert-separator
     "TAB" corfu-next
     "S-TAB" corfu-previous
     "C-j" corfu-next
     "C-k" corfu-previous))

  ;; init
  (global-corfu-mode) ;; グローバルに corfu-mode を有効にする。
  (corfu-popupinfo-mode) ;; extensions に含まれる、補完に説明表示を追加する。default ではオフ。
  )
#+end_src
