* setup.el のブートストラップ

twist.nix がすでにすべてのパッケージを Nix store 上にビルド・配置して
いるので、setup.el 標準の =:package= は何もしない =no-op= に置き換える。

#+begin_src emacs-lisp
;; -*- lexical-binding: t; -*-

(eval-and-compile
  (require 'setup))

(setup-define :package
  (lambda (_order) nil)
  :documentation "no-op: パッケージのインストールは twist.nix が担当する。"
  :repeatable t
  :shorthand #'cadr)
#+end_src

** =setopt= を =:opt= で使えるようにする
#+begin_src emacs-lisp
(setup-define :opt
  (lambda (&rest pairs)
    `(setopt ,@pairs))
  :after-loaded t)
#+end_src
** =:load-after= を使えるようにする
#+begin_src emacs-lisp
  (setup-define :load-after
  (lambda (&rest features)
    (let ((body `(require ',(setup-get 'feature))))
      (dolist (feature (nreverse features))
        (setq body `(with-eval-after-load ',feature ,body)))
      body))
  :documentation "Load the current feature after FEATURES.")
#+end_src
** =:needs= を使えるようにする
#+begin_src emacs-lisp
  (setup-define :needs
  (lambda (executable)
    `(unless (executable-find ,executable)
       ,(setup-quit)))
  :documentation "If EXECUTABLE is not in the path, stop here."
  :repeatable 1)
#+end_src

  (setup-define :global-bind
    (lambda (key command)
      `(keymap-global-set ,key #',command))

** =:mode-remap= を使えるようにする
#+begin_src emacs-lisp
    (setup-define :mode-remap
    (lambda (old-mode)
      `(add-to-list 'major-mode-remap-alist
                     (cons ',old-mode ',(setup-get 'mode))))
    :documentation "Remap OLD-MODE to the current mode via `major-mode-remap-alist'.
  This causes Emacs to automatically use the current mode instead of
  OLD-MODE whenever OLD-MODE would otherwise be activated (e.g. by
  `auto-mode-alist' or `set-auto-mode'). Useful for redirecting legacy
  major modes to their tree-sitter (or other) replacements without
  touching `auto-mode-alist' itself."
    :repeatable t
    :signature '(OLD-MODE ...))

** vertico
#+begin_src emacs-lisp
  (setup (:package vertico)
    (:opt vertico-resize nil
          vertico-count 20
          vertico-cycle t)
    (:with-map vertico-map
      ;; ディレクトリ移動時にディレクトリ単位で移動できる。
      (:bind "DEL" vertico-directory-delete-char))

    (vertico-mode)

    ;; verticoの選択行を指す指アイコンを追加する設定
    (:require nerd-icons)
    (defvar +vertico-current-arrow t)

    (defun +vertico-format-candidate-arrow-advice (orig-fun cand prefix suffix index start)
      "Vertico の選択行に指アイコンを表示するアドバイス"
      (let ((res (funcall orig-fun cand prefix suffix index start)))
        (if (and +vertico-current-arrow 
                 (not (bound-and-true-p vertico-flat-mode)))
            (let ((arrow (nerd-icons-faicon "nf-fa-hand_o_right")))
              (if (bound-and-true-p vertico-grid-mode)
                  (if (= vertico--index index)
                      (concat arrow " " res)
                    (concat "  " res))
                (if (= vertico--index index)
                    (concat " " arrow " " res)
                  (concat "    " res))))
          res)))

    (advice-add 'vertico-format-candidate :around #'+vertico-format-candidate-arrow-advice))
#+end_src
