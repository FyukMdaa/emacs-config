* GC を起動時だけ緩める

起動中は GC をほぼ止めておき、起動が終わったら現実的な値に戻す。

#+begin_src emacs-lisp
;; -*- lexical-binding: t; -*-

(setq gc-cons-threshold most-positive-fixnum
      gc-cons-percentage 0.6)

(add-hook 'emacs-startup-hook
          (lambda ()
            (setq gc-cons-threshold (* 16 1024 1024)
                  gc-cons-percentage 0.1)))
#+end_src

=file-name-handler-alist= も起動中だけ空にしておく。TRAMP 等がこの間は使えなくなる点だけ注意。

#+begin_src emacs-lisp
(defvar my/default-file-name-handler-alist file-name-handler-alist)
(setq file-name-handler-alist nil)

(add-hook 'emacs-startup-hook
          (lambda ()
            (setq file-name-handler-alist my/default-file-name-handler-alist)))
#+end_src

* package.el を無効化する

#+begin_src emacs-lisp
(setq package-enable-at-startup nil)
(setq package-quickstart nil)
#+end_src

* 不要なUIを削除

#+begin_src emacs-lisp
(setq default-frame-alist
      (append '((menu-bar-lines . 0)
                (tool-bar-lines . 0)
                (vertical-scroll-bars . nil))
              default-frame-alist))

(setq inhibit-splash-screen t
      inhibit-startup-message t
      inhibit-startup-echo-area-message user-login-name)
#+end_src

* native-comp まわり

twist.nix がビルド時にパッケージを (ネイティブ) コンパイルするので、
起動時の非同期コンパイルの警告は静かにしておく。

#+begin_src emacs-lisp
(when (boundp 'native-comp-async-report-warnings-errors)
  (setq native-comp-async-report-warnings-errors 'silent))
#+end_src
